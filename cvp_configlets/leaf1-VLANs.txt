!
! Tenant TENANT-1 VLANs
!
vlan 10
   name Data_Network
!
vlan 20
   name Voice_Network
!
interface Vlan10
   description Data_Network
   vrf TENANT-1
   ip address virtual 10.10.10.1/24
!
interface Vlan20
   description Voice_Network
   vrf TENANT-1
   ip address virtual 10.20.20.1/24
!
router bgp 65101
   !
   vlan 10
      rd auto
      route-target both 10:10
      redistribute learned
   !
   vlan 20
      rd auto
      route-target both 20:20
      redistribute learned
!
interface Vxlan1
   vxlan vlan 10 vni 10
   vxlan vlan 20 vni 20
